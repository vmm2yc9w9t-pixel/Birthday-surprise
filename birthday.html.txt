/* 🎂 CANVAS SETUP */
const canvas = document.getElementById("canvas");
const ctx = canvas.getContext("2d");
canvas.width = innerWidth;
canvas.height = innerHeight;

let particles = [];
let mode = "fire";

/* 🔥 FIRE */
function fireEffect() {
  for (let i = 0; i < 5; i++) {
    particles.push({
      x: Math.random()*canvas.width,
      y: canvas.height,
      size: Math.random()*5+2,
      speedY: Math.random()*-3-1,
      color: `hsl(${Math.random()*50},100%,50%)`
    });
  }

  particles.forEach((p,i)=>{
    p.y += p.speedY;
    p.size *= 0.96;

    ctx.beginPath();
    ctx.arc(p.x,p.y,p.size,0,Math.PI*2);
    ctx.fillStyle = p.color;
    ctx.fill();

    if(p.size<0.5) particles.splice(i,1);
  });
}

/* ❤️ HEART */
function heartEffect() {
  for(let t=0;t<Math.PI*2;t+=0.2){
    let x = 16*Math.pow(Math.sin(t),3);
    let y = -(13*Math.cos(t)-5*Math.cos(2*t)-2*Math.cos(3*t)-Math.cos(4*t));

    ctx.fillStyle="red";
    ctx.fillRect(canvas.width/2+x*10,canvas.height/2+y*10,2,2);
  }
}

/* 💌 STORY */
let lines = [
"From the moment I met you… 💫",
"You became my Penguin 🐧",
"My cute little Cat 🐱",
"My everything 💖",
"Today you turn 22 ✨",
"And I still fall for you every day ❤️",
"Happy Birthday Zaibun 🔥"
];

let i=0;

function showStory(){
  if(i<lines.length){
    document.getElementById("story").innerHTML = lines[i];
    document.getElementById("story").style.opacity = 1;
    speak(lines[i]);
    i++;
    setTimeout(showStory,2500);
  } else {
    document.getElementById("photo").style.display = "block";
    document.getElementById("cake").style.display = "block";
    mode="heart";
  }
}

/* 🔊 VOICE */
function speak(text){
  let speech = new SpeechSynthesisUtterance(text);
  speech.rate = 0.9;
  speech.pitch = 1.2;
  speech.lang = "en-US";
  speechSynthesis.speak(speech);
}

/* 🎂 CAKE */
function cutCake(){
  alert("🎂 Happy Birthday My Love 💖🔥");
}

/* 🎞 LOOP */
function animate(){
  ctx.clearRect(0,0,canvas.width,canvas.height);

  if(mode==="fire") fireEffect();
  if(mode==="heart") heartEffect();

  requestAnimationFrame(animate);
}
animate();

/* 🚀 START */
function startExperience(){
  showStory();
}
